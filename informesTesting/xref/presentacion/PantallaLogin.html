199 		PersonalVicerrectorado personalVicerrectorado = pv1.seleccionarProfesor(nombreUsuario);
200 		if (personalVicerrectorado.getPassword().compareTo(contrasena) == 0) {
201 			if (personalVicerrectorado.isJefe() == true) {
202 				PantallaJefeVicerrectorado pjfGabineteVicerrectorado = new PantallaJefeVicerrectorado();
203 				pjfGabineteVicerrectorado.setVisible(true);
204 			} else {
