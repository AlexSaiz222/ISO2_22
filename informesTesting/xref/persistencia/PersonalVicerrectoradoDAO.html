30  	public int crearVicerrectorado(PersonalVicerrectorado vicerrectorado) throws SQLException {
31  		int resultado = -1;
32  		GestorBD agente = new GestorBD();
33  
34  		PreparedStatement pstmt = null;
35  		try {
36  			pstmt = agente.mBD.prepareStatement(
37  					"insert into vicerrectorado (dni, nombre, password, apellidos, doctor) " + "values (?,?,?,?,?)");
38  			pstmt.setString(1, vicerrectorado.getDni());
39  			pstmt.setString(2, vicerrectorado.getNombre());
40  			pstmt.setString(3, vicerrectorado.getApellidos());
41  			pstmt.setString(4, vicerrectorado.getNombre() + vicerrectorado.getApellidos());
42  			pstmt.setBoolean(5, vicerrectorado.isJefe());
43  
44  			resultado = agente.insert(pstmt);
45  
46  		} catch (SQLException e) {
47  			System.out.println("PersonalVicerrectoradorDAO: " + e.getMessage());
48  		} finally {
49  			if (pstmt != null)
50  				pstmt.close();
51  		}
52  
53  		return resultado;
54  	}
